createSord: function(params)
{
    if(!params)
    {
        throw "params are empty";
    }

    var parentId = params.parentId || "1";
    var maskId = params.maskId || "1";
    var sord = ixConnect.ix().createSord(parentId, maskId, EditInfoC.mbSord).sord;
    sord.name = params.name || "unnamed";

}
